    asticaVision - Image Description Usage:
    
        
    <script src="https://www.astica.org/endpoint/ml/javascript/latest/astica.api.js"></script>
<script>
    asticaAPI_start('API KEY HERE');
    asticaVision_analyze(
        '1.0_full', //modelVersion: 1.0_full, 2.0_full
        'IMAGE URL', //Input Image
        'Description, Faces' //Description, Moderate, Faces Celebrities, Landmarks
        asticaVision_result, //Callback function
    );      
    function asticaVision_result(data) { console.log(data); } //receive output
    function asticaVision_error(data)  { console.log(data); } //catch error
</script>
